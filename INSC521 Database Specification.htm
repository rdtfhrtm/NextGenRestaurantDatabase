USERDOB DATE,

SELECT "USERNAME", (EXTRACT(YEAR FROM SYSDATE) - EXTRACT(YEAR FROM USERDOB)) AS AGE

FROM P_USER PU, P_STAFF PS

WHERE PS.USERID = PU.USERID AND STAFFROLE = 'SERVER'

AND (EXTRACT(YEAR FROM SYSDATE) - EXTRACT(YEAR FROM USERDOB)) > 21

ORDER BY AGE;
